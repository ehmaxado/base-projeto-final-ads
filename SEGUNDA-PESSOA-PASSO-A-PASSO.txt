CODIGO EXEMPLO PARA A TELA DE PRODUTOS

Arquivo sugerido:

- frontend/app/(system)/produtos/actions.ts

Exemplo de action para criar produto:

```tsx
'use server'

import { revalidatePath } from 'next/cache'
import { apiServerFetch } from '@/lib/api-server'

export async function criarProduto(formData: FormData) {
  const payload = {
    nome: String(formData.get('nome') ?? ''),
    codigoSku: String(formData.get('codigoSku') ?? ''),
    descricao: String(formData.get('descricao') ?? ''),
    precoUnitario: Number(formData.get('precoUnitario') ?? 0),
    unidade: String(formData.get('unidade') ?? 'UN'),
    ativo: formData.get('ativo') === 'on',
  }

  const response = await apiServerFetch('/produtos', {
    method: 'POST',
    body: JSON.stringify(payload),
  })

  if (!response.ok) {
    throw new Error('Erro ao criar produto')
  }

  revalidatePath('/produtos')
}
```

Exemplo de tipo para produto:

```ts
export type Produto = {
  id: number
  codigoSku: string | null
  nome: string
  descricao: string | null
  precoUnitario: number
  unidade: string
  ativo: boolean
}
```

Exemplo de busca real na page:

```tsx
const response = await apiServerFetch('/produtos', { cache: 'no-store' })
const produtos = response.ok ? ((await response.json()) as Produto[]) : []
```

Exemplo para atualizar perfil:

```tsx
'use server'

import { revalidatePath } from 'next/cache'
import { apiServerFetch } from '@/lib/api-server'

export async function atualizarUsuario(formData: FormData) {
  const payload = {
    nomeCompleto: String(formData.get('nomeCompleto') ?? ''),
    email: String(formData.get('email') ?? ''),
  }

  const response = await apiServerFetch('/usuarios/atual', {
    method: 'PATCH',
    body: JSON.stringify(payload),
  })

  if (!response.ok) {
    throw new Error('Erro ao atualizar usuario')
  }

  revalidatePath('/usuario')
}
```

Exemplo para atualizar senha:

```tsx
'use server'

import { apiServerFetch } from '@/lib/api-server'

export async function atualizarSenha(formData: FormData) {
  const novaSenha = String(formData.get('novaSenha') ?? '')
  const confirmarSenha = String(formData.get('confirmarSenha') ?? '')

  if (novaSenha !== confirmarSenha) {
    throw new Error('As senhas nao conferem')
  }

  const payload = {
    senhaAtual: String(formData.get('senhaAtual') ?? ''),
    novaSenha,
  }

  const response = await apiServerFetch('/usuarios/atual/senha', {
    method: 'PATCH',
    body: JSON.stringify(payload),
  })

  if (!response.ok) {
    throw new Error('Erro ao atualizar senha')
  }
}
```

Exemplo de tipo para usuario:

```ts
export type UsuarioAtual = {
  id: number
  nomeCompleto: string
  email: string
  perfil: string
  ativo: boolean
}
```
